// fetched data apiCall = null; // apiCall and its cached data can now be garbage-collected ### WeakSet #### Definition A `WeakSet` is a collection of objects, where an object can only appear once. The main characteristic of `WeakSet` is that it holds "weak" references to its objects, meaning that if there are no other references to an object stored in the `WeakSet`, it can be garbage-collected. This makes `WeakSet` suitable for keeping track of objects without preventing them from being garbage-collected. #### Use Cases 1. **Additional Data** - Tracking objects without needing to keep them alive solely for the tracking purpose. 2. **Caching** - Remembering objects that you have seen or processed in some way, without affecting their lifecycle. #### Example Usage 1. **Tracking Processed Objects** let processed = new WeakSet(); function process(obj) { if (processed.has(obj)) { console.log('Already processed:', obj); return; } // Process the object processed.add(obj); } let item1 = { id: 1 }; let item2 = { id: 2 }; process(item1); // Processed process(item1); // Already processed item1 = null; // item1 can now be garbage-collected 2. **Avoiding Reprocessing** let seen = new WeakSet(); function handleObject(obj) { if (seen.has(obj)) { return; // Skip if already seen } seen.add(obj); // Perform operations on obj } let obj1 = { name: 'Widget' }; let obj2 = { name: 'Gadget' }; handleObject(obj1); handleObject(obj1); // Will not process again handleObject(obj2); obj1 = null; // obj1 can now be garbage-collected #### Practice 1. **Add and Check Objects in WeakSet** let ws = new WeakSet(); let obj1 = { id: 1 }; let obj2 = { id: 2 }; let obj3 = { id: 3 }; ws.add(obj1); ws.add(obj2); ws.add(obj3); console.log(ws.has(obj1)); // true console.log(ws.has(obj2)); // true console.log(ws.has(obj3)); // true 2. **Nullify Object and Check WeakSet** let ws = new WeakSet(); let obj = { id: 1 }; ws.add(obj); console.log(ws.has(obj)); // true obj = null; // obj is now eligible for garbage collection // After a while, obj will be removed from the WeakSet 3. **Tracking Processed Objects** let processedObjects = new WeakSet(); function processObject(obj) { if (processedObjects.has(obj)) { console.log('Object already processed:', obj); return; } console.log('Processing object:', obj); processedObjects.add(obj); } let object1 = { name: 'Object1' }; let object2 = { name: 'Object2' }; processObject(object1); // Processing object processObject(object1); // Object already processed processObject(object2); // Processing object object1 = null; // object1 can now be garbage-collected 4. **Avoid Duplicate Processing** let seenObjects = new WeakSet(); function processUniqueObject(obj) { if (seenObjects.has(obj)) { console.log('Duplicate object, skipping:', obj); return; } console.log('Processing unique object:', obj); seenObjects.add(obj); } let obj1 = { type: 'Type1' }; let obj2 = { type: 'Type2' }; processUniqueObject(obj1); // Processing unique object processUniqueObject(obj1); // Duplicate object, skipping processUniqueObject(obj2); // Processing unique object obj1 = null; // obj1 can now be garbage-collected ### Summary - **WeakMap** and **WeakSet** are special types of collections that allow objects to be garbage-collected if they are no longer referenced anywhere else in the application. - **WeakMap** is used for key/value pairs with object keys, while **WeakSet** is used for storing unique objects. - They are particularly useful for attaching additional data to objects or for caching purposes without affecting the lifecycle of the objects.
